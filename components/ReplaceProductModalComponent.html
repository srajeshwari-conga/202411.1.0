Description

The ReplaceProductModalComponent is used to display a modal for replacing products based on replacement constraint rules. To use this component, the rules input property should be provided with the relevant constraint rule groups.

Preview

Usage

import { Component } from '@angular/core';
import { ConstraintRuleGroups } from '../../constraint-rule/interfaces/constraint-rule.interface';
import { ReplaceProductModalComponent } from './replace-product-modal.component';

@Component({
  ...
})
export class AppComponent {

  replaceProductModal: ReplaceProductModalComponent;

  openReplaceModal() {
    this.replaceProductModal.openModal();
  }
}
